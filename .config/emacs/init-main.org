    "Add DIR as a .pth file to the site-packages of the current pyvenv environment (Unix only)."
    (interactive "DDirectory to add to venv Python path (as .pth): ")
    (unless (and (boundp 'pyvenv-virtual-env) pyvenv-virtual-env)
